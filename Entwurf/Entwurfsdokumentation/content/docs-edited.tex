	\texdocfield{private}{String}{discipline}{Das Fach des Moduls}
	\texdocfield{private}{String}{lecturer}{Der Dozent des Moduls}
	\texdocfield{private}{ModuleConstraint}{moduleconstraint}{Die zum Modul gehörenden System-Constraints}
	\texdocfield{private}{String}{moduleId}{Die ID des Moduls}
	\texdocfield{private}{String}{name}{Der Name des Moduls}
	\texdocfield{private}{boolean}{passed}{Ob das Modul bereits vom Nutzer bestanden wurde. Dies wird gespeichert,
		da die Trennung zwischen Modulen im Studienplan und bestandenen Modulen
		für die Anzeige der Module auf der Client-Seite aufgehoben wird. Dies ist
		problemlos möglich, da diese Trennung lediglich für Berechnungen
		interessant sind, welche nicht auf der Client-Seite stattfinden.}
	\texdocfield{private}{String}{planId}{Die ID des Plans zu der das Modul gehört (da die Präferenzen
